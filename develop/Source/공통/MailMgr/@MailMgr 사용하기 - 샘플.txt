// Mail Manger Dialog를 띄우고 보낼 메일을 체크하여 보낼 때
///////////////////////////////////////////////////////////////////////////////

void CMailMgrTestDlg::OnbMailTest() 
{
	///Mail Manager 객체 생성
	CSendMailMgr MailMgr(this);

	///Mail Record 객체 생성
	SendMailRecord MailRecord;

	///Mail Record 객체 설정
	MailRecord.m_nFileAttachCnt = 0;
	MailRecord.m_pFileAttach = "";
	MailRecord.m_sBody = "<head><title>첫번째</title></head><body>첫번째 내용</body>";
	MailRecord.m_sEncoding = "text/html";
	MailRecord.m_sRecipientAddress = "gjpark@eco.co.kr";	//전자우편
	MailRecord.m_sRecipientName	 = "박길주";	//신청자이름
	MailRecord.m_sSubject = "CIP 반송 - 첫번째";
	MailRecord.m_sCCName    = "trzeze";
	MailRecord.m_sCCAddress = "trzeze@eco.co.kr";
	
	///Mail Record 객체 추가
	MailMgr.AddMailRecord(&MailRecord);

	///Mail Record 객체 설정
	MailRecord.m_nFileAttachCnt = 0;
	MailRecord.m_pFileAttach = "";
	MailRecord.m_sBody = "<head><title>두번째</title></head><body>두번째 내용</body>";
	MailRecord.m_sEncoding = "text/html";
	MailRecord.m_sRecipientAddress = "gjpark@eco.co.kr";	//전자우편
	MailRecord.m_sRecipientName	 = "박길주";	//신청자이름
	MailRecord.m_sSubject = "CIP 반송 - 두번째";
	MailRecord.m_sCCName    = "trzeze";
	MailRecord.m_sCCAddress = "trzeze@eco.co.kr";
	
	///Mail Record 객체 추가
	MailMgr.AddMailRecord(&MailRecord);

	///Mail Manager Dialog 뛰우기
	MailMgr.DoModal();

	///결과 확인
	///전송 성공한 메일의 개수
	INT nCntSended = MailMgr.GetCountMailSended();
	///전송 실패한 메일의 개수
	INT nCntFailed = MailMgr.GetCountMailFailed();
}
